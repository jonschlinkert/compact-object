## Usage

```js
var compact = require('{%= name %}');

compact({a: 'a', b: ''});
//=> {a: 'a'}

compact({a: 'a', b: {c: 'c', d: ''});
//=> {a: 'a', b: {c: 'c'}

compact({a: ['a'], b: [['foo'], [], ['baz']]});
//=> {a: ['a'], b: ['foo', 'baz']}

compact({a: 0, b: 1});
//=> {a: 0, b: 1}

// set the `noZero` flag
compact({a: 0, b: 1}, true);
//=> {b: 1}
```
